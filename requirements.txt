# PerceptNet core requirements (installs clean on macOS arm64 / Python 3.11, CPU/MPS).
#
# Install:  python3.11 -m venv .venv && source .venv/bin/activate && pip install -r requirements.txt
# Or, preferred:  pip install -e ".[dev]"
#
# CUDA-only and ROS-only dependencies are intentionally NOT here:
#   - GPU (TensorRT, pycuda, onnxruntime-gpu)  -> requirements-gpu.txt
#   - ROS 2 (rclpy is apt-only, vision_msgs)   -> requirements-ros.txt

# --- deep learning ---
torch>=2.2,<3.0

# --- camera / vision ---

ultralytics>=8.1,<8.3          # pinned tight: feature-tap internals churn across minors

# --- point cloud + 3D viz ---
open3d>=0.18                   # first reliable macOS arm64 wheels

# --- model export / inference ---
onnx>=1.15
onnxruntime>=1.17              # CPU build; do NOT also install onnxruntime-gpu (conflict)

# --- experiment tracking + io ---
mlflow>=2.10,<3.0
